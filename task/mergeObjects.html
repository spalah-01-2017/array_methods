Если имена свойств двух объектов совпапдают, значение соответсвующего свойства возвращаемого объекта должно быть равным значению свойства второго объекта.

Пример:

var obj1 = { a: 10, b: 20, c: 30 };
var obj2 = { q: 8, b: 7, c: 6 };

mergeObjects(obj1,obj2);
// {
// a: 10,
// b: 7,
// c: 6,
// q: 8
